/**
 * Actions
 * @enum {string}
 */
var Actions = {
    /** reply to the message */
    REPLY:'REPLY',
    /** terminate the message */
    CLOSE:'CLOSE',
    /** drop off the message */
    LEAVE:'LEAVE',
    ACCEPT:'ACCEPT',
    REJECT:'REJECT',
    OK:'OK',
    /** add a member to the message */
    FORWARD:'FORWARD',
    /** substitute another member for yourself in the message */
    DELEGATE:'DELEGATE'
};
